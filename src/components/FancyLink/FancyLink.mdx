# FancyLink

<Playground>
  {() => {
    const handleClick = (event) => event.preventDefault();
    return (
      <div className="fancy-link">
        {/* Use `href` prop to link to external sites */}
        <FancyLink onClick={handleClick} href="https://github.com/hellobrian">
          This uses href
        </FancyLink>
        {/* Or use `to` prop to use Gatsby Link. */}
        <FancyLink onClick={handleClick} to="/">

        </FancyLink>
      </div>
    );
  }}
